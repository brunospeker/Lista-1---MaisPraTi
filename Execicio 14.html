Crie um programa que calcula o fatorial de um número fornecido pelo usuário utilizando um loop for ou while.

Saidas

O resultado aparecerá aqui.

function fatorial() {
    let e1 = document.getElementById("entrada1").value;
    let fat = 1;
    let contador = e1;
    
    while( contador > 1){
        fat *= contador;
        contador--;
        console.log(fat);
    }
    console.log("O Fatorial de "+e1+" é "+fat+"!")
    document.getElementById("resultado").innerText = "O Fatorial de "+e1+" é "+fat+"!";
}
