function calculaIMC() {
    let peso = parseFloat(document.getElementById("entrada1").value);
    let altura = parseFloat(document.getElementById("entrada2").value);
    let imc = peso / (altura * altura)
    console.log(imc)
    if(imc < 18){
        document.getElementById("resultado").innerText = "O seu IMC é de "+imc+" e esá classificado como Baixo Peso!";
        console.log("O seu IMC é de "+imc+" e está classificado como Baixo Peso!");
    } else if(imc >= 18 && imc < 25){
        document.getElementById("resultado").innerText = "O seu IMC é de "+imc+" e esá classificado como Normal!";
        console.log("O seu IMC é de "+imc+" e está classificado como Normal!");
    } else if(imc >= 25 && imc < 30){
        document.getElementById("resultado").innerText = "O seu IMC é de "+imc+" e esá classificado como Sobrepeso!";
        console.log("O seu IMC é de "+imc+" e está classificado como Sobrepeso!");
    } else{
        document.getElementById("resultado").innerText = "O seu IMC é de "+imc+" e esá classificado como Obesidade!";
        console.log("O seu IMC é de "+imc+" e está classificado como Obesidade!");
    }
