function compararDoisNumeros() {
    let e1 = parseFloat(document.getElementById("entrada1").value);
    let e2 = parseFloat(document.getElementById("entrada2").value);
    if(e1 != e2){
        if(e1 < e2){
            document.getElementById("resultado").innerText = "Primeiro vem o valor "+e1+" e depois o valor "+e2+"!";
            console.log(e1);
            console.log("Primeiro vem o valor "+e1+" e depois o valor "+e2+"!");
        } else {
            document.getElementById("resultado").innerText = "Primeiro vem o valor "+e2+" e depois o valor "+e1+"!";
            console.log(e1);
            console.log("Primeiro vem o valor "+e2+" e depois o valor "+e1+"!");
        }
    } else{
        document.getElementById("resultado").innerText = "Os valores são iguais. Por favor, coloque outros valores.";
        console.log("Os valores são iguais. Por favor, coloque outros valores.");

    }
}
