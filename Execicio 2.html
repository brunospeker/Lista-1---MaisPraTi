Crie um programa que classifica a idade de uma pessoa em categorias (criança, adolescente, adulto, idoso) com base no valor fornecido, utilizando uma estrutura de controle if-else.

Saidas

function verificaIdade() {
    let e1 = document.getElementById("entrada1").value;
    console.log(e1);
    if(e1 <= 12){
        document.getElementById("resultado").innerText = "Uma pessoa com "+e1+" anos é classificada como Criança!";
        console.log("Uma pessoa com "+e1+" anos é classificada como Criança!");
    } else if( e1 >= 13 && e1 <= 17 ) {
        document.getElementById("resultado").innerText = "Uma pessoa com "+e1+" anos é classificada como Adolescente!";
        console.log("Uma pessoa com "+e1+" anos é classificada como Adolescente!");
    } else if( e1 >= 18 && e1 <= 64 ) {
        document.getElementById("resultado").innerText = "Uma pessoa com "+e1+" anos é classificada como Adulto!";
        console.log("Uma pessoa com "+e1+" anos é classificada como Adulto!");
    } else if( e1 >= 65) {
        document.getElementById("resultado").innerText = "Uma pessoa com "+e1+" anos é classificada como Idoso!";
        console.log("Uma pessoa com "+e1+" anos é classificada como Idoso!");
    }
}
