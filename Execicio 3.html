function verificaNota() {
    let e1 = document.getElementById("entrada1").value;
    console.log(e1);
    if(e1 >= 7){
        document.getElementById("resultado").innerText = "O aluno recebeu a nota "+e1+" e foi Aprovado!";
        console.log("O aluno recebeu a nota "+e1+" e foi Aprovado!");
    } else if( e1 >= 5 && e1 < 7 ) {
        document.getElementById("resultado").innerText = "O aluno recebeu a nota "+e1+" e ficou de Recuperação!";
        console.log("O aluno recebeu a nota "+e1+" e ficou de Recuperação!");
    } else if( e1 < 5) {
        document.getElementById("resultado").innerText = "O aluno recebeu a nota "+e1+" e foi Reprovado!";

}
