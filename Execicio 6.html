Exercício 6

Ler três valores para os lados de um triângulo: A, B e C. Verificar se os lados fornecidos formam realmente um triângulo. Caso forme, deve ser indicado o tipo de triângulo: Isósceles, escaleno ou eqüilátero.

Saidas

function verificaTriangulo() {
    let lado1 = parseFloat(document.getElementById("entrada1").value);
    let lado2 = parseFloat(document.getElementById("entrada2").value);
    let lado3 = parseFloat(document.getElementById("entrada3").value);

    if(lado1 < (lado2 + lado3) && lado2 < (lado1 + lado3) && lado3 < (lado2 + lado1)){
        if(lado1 === lado2 && lado2 === lado3){
            document.getElementById("resultado").innerText = "Os três lados formam um Triângulo Equilátero.";
            console.log("Os três lados formam um Triângulo Equilátero.");
        }else if(lado1 === lado2 || lado1 === lado3 || lado2 === lado3){
            document.getElementById("resultado").innerText = "Os três lados formam um Triângulo Isósceles.";
            console.log("Os três lados formam um Triângulo Isósceles.");
        }  else{
            document.getElementById("resultado").innerText = "Os três lados formam um Triângulo Escaleno.";
            console.log("Os três lados formam um Triângulo Escaleno.");
        }
    } else {
        document.getElementById("resultado").innerText = "Não é um triângulo.";
        console.log("Não é um triângulo.");
    }
}
