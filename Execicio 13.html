let contagem = [];

function mediaAritmetica() {
    let e1 = parseFloat(document.getElementById("entrada1").value);
    console.log(contagem)
    if(e1 === 0){
        quantidade = 0;
        soma = 0;

        for(let c of contagem){
            soma += c;
            quantidade += 1;
        }
        mediafinal = soma / quantidade;
        console.log(mediafinal)
        console.log("A média aritmética é "+mediafinal+" no array.")
        document.getElementById("resultado").innerText = "A média aritmética é "+ mediafinal +" no array.";
    }else{
        contagem.push(e1);
        console.log("Adicionado o número "+e1+" no array.")
        document.getElementById("resultado").innerText = "Adicionado o número "+e1+" no array.";
    } 
}
