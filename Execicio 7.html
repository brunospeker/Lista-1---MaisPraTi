As maçãs custam R$ 0,30 se forem compradas menos do que uma dúzia, e R$ 0,25 se forem compradas pelo menos doze. Escreva um algoritmo que leia o número de maçãs compradas, calcule e escreva o valor total da compra.

function contarMacas() {
    let e1 = parseFloat(document.getElementById("entrada1").value);
    console.log(e1)
    if(e1 < 12){
        let preco = (0.30 * e1).toFixed(2);
        document.getElementById("resultado").innerText = "O custo total foi de R$"+preco+" reais, pois foram compradas "+e1+" maçãs a um preço de R$0,30!";
        console.log("O custo total foi de R$"+preco+" reais, pois foram compradas "+e1+" maçãs a um preço de R$0,30!");
    } else {
        let preco = (0.25 * e1).toFixed(2);
        document.getElementById("resultado").innerText = "O custo total foi de R$"+preco+" reais, pois foram compradas "+e1+" maçãs a um preço de R$0,25!";
        console.log("O custo total foi de R$"+preco+" reais, pois foram compradas "+e1+" maçãs a um preço de R$0,25!");
    }
}
