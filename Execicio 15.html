Escreva um programa que gera e imprime os primeiros 10 números da sequência de Fibonacci utilizando um loop for.

Saidas

function fibonacci() {
    let e1 = parseInt(document.getElementById("entrada1").value);
    let a = 0;
    let b = 1;
    let proximo = 0;
    console.log("O 1º número é o "+a)
    console.log("O 2º número é o "+b)
    
    for (let i = 2; i <= e1; i++) {
        proximo = a + b;
        a = b;
        b = proximo;
        console.log("O "+i+"º número é o "+proximo)
    }
}
