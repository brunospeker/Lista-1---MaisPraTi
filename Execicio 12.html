Crie um programa que exibe a tabuada de um número fornecido pelo usuário (de 1 a 10) utilizando um loop for.

Saidas

function tabuada() {
    let e1 = parseInt(document.getElementById("entrada1").value);
    console.log(e1);
    for(let n = 0; n <= 10; n++){
        multiplicacao = n * e1;
        document.getElementById("resultado").innerText = "O número é "+e1+" multiplicado por "+n+" é igual a "+multiplicacao+"! Verifique no console para enxergar a contagem.";
        console.log("O número é "+e1+" multiplicado por "+n+" é igual a "+multiplicacao+"! Verifique no console para enxergar a contagem.")
    }
}
